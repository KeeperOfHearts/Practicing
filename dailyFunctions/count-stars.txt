class CountStars {
    public static int[] countStars(int[] sky, int maxCount) {
        int[] result = new int[sky.length];
        boolean max = false;
        int count = 0;
        for (int i = 0;i < sky.length;i++) {
            if (max) {
                result[i] = 0;
                continue;
            }
            result[i] = sky[i];
            if (sky[i] == 1) count++; 
            if (count == maxCount) max = true;
        }
        return result;
    }
}
