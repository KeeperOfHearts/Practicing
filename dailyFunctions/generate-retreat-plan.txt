class GenerateRetreatPlan {
    public static String[] generateRetreatPlan(String[] guestLogs, int maxSupplies) {
        int length = (guestLogs.length > maxSupplies) ? maxSupplies : guestLogs.length;
        if (length > 24) length = 24;
        String[] result = new String[length];
        for (int i = 0;i < result.length;i++) {
            String[] values = guestLogs[i].split(":");
            result[i] = String.format("%s arrives at %s:%s", values[0], Integer.toBinaryString(Integer.parseInt(values[1])), values[2]);
        }
        return result;
    }
}
