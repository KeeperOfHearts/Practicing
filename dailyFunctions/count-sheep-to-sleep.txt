class CountSheepToSleep {
    public static String countSheepToSleep(int sheepCount, int distractionFactor) {
        int result = 0;
        for (int sheep = 1;sheep<=sheepCount;sheep++) {
            if (sheep % distractionFactor != 0) result++;
            if (sheep % 7 == 0) break;
        }
        return String.format("You fell asleep after counting %d sheep.", result);
    }
}
